#!/bin/bash
# EasePi A2 OLED & WiFi 一键初始化脚本
# 自动安装依赖/配置服务/激活WiFi驱动
# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'
# 全局变量
SCRIPT_DIR="/usr/local/oled"
PY_SCRIPT="${SCRIPT_DIR}/custom_oled.py"
SERVICE_FILE="/etc/systemd/system/oled.service"
INIT_SERVICE="/etc/systemd/system/oled-init.service"
LOG_DIR="/var/log/oled"
LOG_FILE="${LOG_DIR}/oled.log"
INIT_MARKER="${SCRIPT_DIR}/.init_completed"
# 提前创建日志目录
[ ! -d "${LOG_DIR}" ] && mkdir -p "${LOG_DIR}" > /dev/null 2>&1
touch "${LOG_FILE}" > /dev/null 2>&1
chmod 644 "${LOG_FILE}" > /dev/null 2>&1
log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
    echo "$(date +'%Y-%m-%d %H:%M:%S') - INFO - $1" >> ${LOG_FILE} 2>&1
}
log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
    echo "$(date +'%Y-%m-%d %H:%M:%S') - WARN - $1" >> ${LOG_FILE} 2>&1
}
log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
    echo "$(date +'%Y-%m-%d %H:%M:%S') - ERROR - $1" >> ${LOG_FILE} 2>&1
}
# 检查是否已经初始化完成
check_init_status() {
    if [ -f "${INIT_MARKER}" ]; then
        log_info "初始化已经完成，跳过执行。"
        exit 0
    fi
}
# 1. 根权限校验
check_root() {
    if [ "$EUID" -ne 0 ]; then
        log_error "请以root权限运行此脚本"
        exit 1
    fi
}
# 2. 检查脚本文件完整性
check_files() {
    log_info "检查脚本目录与文件完整性"
    if [ ! -d "${SCRIPT_DIR}" ]; then
        mkdir -p "${SCRIPT_DIR}"
        chmod 755 "${SCRIPT_DIR}"
    fi
    [ ! -f "${PY_SCRIPT}" ] && { log_error "核心脚本缺失：${PY_SCRIPT}"; exit 1; }
    [ ! -f "${SERVICE_FILE}" ] && { log_error "服务文件缺失：${SERVICE_FILE}"; exit 1; }
   
    chmod +x "${PY_SCRIPT}"
    chmod +x "${0}"
}
# 3. 安装系统依赖
install_deps() {
    log_info "安装 OLED 依赖"

    if command -v timedatectl >/dev/null 2>&1; then
        timedatectl set-ntp true >/dev/null 2>&1
    fi

    # 更新软件源
    if command -v apt-get >/dev/null 2>&1; then
        log_info "刷新软件包索引"
        apt-get update \
            -o Acquire::Check-Valid-Until=false \
            >> ${LOG_FILE} 2>&1

        deps=(
            python3
            python3-pip
            python3-smbus
            i2c-tools
        )

        for pkg in "${deps[@]}"; do
            if ! dpkg -s "$pkg" >/dev/null 2>&1; then
                log_info "安装系统包: $pkg"
                apt-get install -y \
                    --no-install-recommends \
                    "$pkg" \
                    >> ${LOG_FILE} 2>&1
            fi
        done
    fi

    log_info "检查 Python 库: smbus2"
    pip3 install smbus2 --break-system-packages --quiet >> ${LOG_FILE} 2>&1

    if [ $? -eq 0 ]; then
        log_info "✓ smbus2 安装成功"
    else
        log_error "✗ smbus2 安装失败，请检查镜像联网状态"
    fi

    log_info "✓ OLED 依赖安装完成"
}

# 4. 安装并激活 WiFi 驱动
install_wifi_driver() {
    log_info "开始配置 WiFi 驱动依赖索引"
   
    if command -v depmod &> /dev/null; then
        depmod -a >> ${LOG_FILE} 2>&1
        log_info "✓ 模块依赖索引已刷新"
    else
        log_error "未找到 depmod 命令"
    fi
    # 尝试加载驱动
    log_info "正在尝试加载 brcmfmac 驱动..."
    modprobe brcmfmac >> ${LOG_FILE} 2>&1
   
    # 确保重启后自动加载（即使脚本不再运行）
    mkdir -p /etc/modules-load.d/
    echo "brcmfmac" > /etc/modules-load.d/brcm.conf
    log_info "✓ 已设置 brcmfmac 开机自动加载"
}
# 5. 检测I2C-3总线
check_i2c() {
    log_info "检测I2C-3总线"
    if ! lsmod | grep -q "i2c_dev"; then
        modprobe i2c-dev >> ${LOG_FILE} 2>&1
    fi
    if [ -e "/dev/i2c-3" ]; then
        chmod 666 /dev/i2c-3
        log_info "✓ i2c-3设备节点就绪"
    else
        log_error "✗ i2c-3设备节点不存在"
        exit 1
    fi
}
# 6. 配置并启动显示服务
config_service() {
    log_info "配置OLED显示服务"
    systemctl daemon-reload >> ${LOG_FILE} 2>&1
    systemctl enable oled.service >> ${LOG_FILE} 2>&1
    log_info "✓ OLED服务已配置完成"
}
# 7. 创建标记文件
disable_self() {
    log_info "创建初始化完成标记"
    touch "${INIT_MARKER}" >> ${LOG_FILE} 2>&1
    chmod 644 "${INIT_MARKER}" >> ${LOG_FILE} 2>&1
   
    if [ -f "${INIT_MARKER}" ]; then
        log_info "✓ 标记文件创建成功"
        systemctl enable oled-init.service >> ${LOG_FILE} 2>&1
    else
        log_error "✗ 标记文件创建失败"
    fi
}
# 8. 显示管理命令
show_commands() {
    echo -e "\n${GREEN}✅ OLED & WiFi 机制部署完成！${NC}"
    echo -e " • WiFi 状态：ip addr show wlan0"
    echo -e " • 核心服务：systemctl status oled.service"
    echo -e " • 初始记录：cat /var/log/oled/oled.log"
    echo -e ""
}
main() {
    clear
    echo -e "${GREEN}=== EasePi A2 OLED & WiFi 初始化流程 ===${NC}"
   
    # 1. 首先检查是否已经完成过初始化
    # 如果已经有标记文件，直接提示并退出，不需要让用户白等30秒
    if [ -f "${INIT_MARKER}" ]; then
        log_info "初始化已经完成，跳过执行。"
        exit 0
    fi
    # 2. 根权限校验（在等待前先确认权限，省得白等）
    check_root
    # 3. 核心延迟等待逻辑
    log_info "等待系统环境就绪 (防止网络未连接或 apt 锁冲突)..."
    for i in {10..1}; do
        # \r 是回车符，能让倒计时在同一行刷新
        echo -ne "\r${YELLOW}[WAIT]${NC} 倒计时 ${i} 秒后正式开始..."
        sleep 1
    done
    echo -e "\n${GREEN}[READY]${NC} 系统环境已就绪，开始安装..."
   
    # 4. 执行后续安装步骤
    check_files
    install_deps
    install_wifi_driver
    check_i2c
    config_service
    disable_self
    show_commands
}
# 脚本最后执行入口
main

OLED_ADDR = 0x3C

PAGES = 4

bus = SMBus(I2C_BUS)

# =========================================================
# SSD1306 LOW LEVEL
# =========================================================

def cmd(c):
    bus.write_byte_data(OLED_ADDR, 0x00, c)

def data(buf):
    for i in range(0, len(buf), 16):
        chunk = [0x40] + list(buf[i:i+16])
        bus.i2c_rdwr(i2c_msg.write(OLED_ADDR, chunk))

def oled_init():

    init_cmds = [
        0xAE,
        0x20, 0x00,
        0xB0,
        0xC8,
        0x00,
        0x10,
        0x40,
        0x81, 0x7F,
        0xA1,
        0xA6,
        0xA8, 0x1F,
        0xA4,
        0xD3, 0x00,
        0xD5, 0x80,
        0xD9, 0xF1,
        0xDA, 0x02,
        0xDB, 0x40,
        0x8D, 0x14,
        0xAF
    ]

    for c in init_cmds:
        cmd(c)

def oled_off():
    cmd(0xAE)

def oled_clear():

    for page in range(PAGES):

        cmd(0xB0 + page)
        cmd(0x00)
        cmd(0x10)

        data(bytearray(128))

# =========================================================
# FRAMEBUFFER
# =========================================================

buffer = bytearray(WIDTH * PAGES)

def clear_buffer():

    global buffer
    buffer = bytearray(WIDTH * PAGES)

def draw_char(x, page, ch):

    font = FONT.get(ch.upper(), FONT[' '])

    for col in range(5):

        if x + col >= WIDTH:
            continue

        buffer[page * WIDTH + x + col] = font[col]

def draw_text(x, page, text):

    for i, c in enumerate(text):
        draw_char(x + i * 6, page, c)

def flush():

    for page in range(PAGES):

        cmd(0xB0 + page)
        cmd(0x00)
        cmd(0x10)

        start = page * WIDTH
        end = start + WIDTH

        data(buffer[start:end])

# =========================================================
# NETWORK
# =========================================================

def detect_iface():

    candidates = [
        'eth0',
        'enp1s0',
        'end0',
        'wlan0'
    ]

    for i in candidates:

        if os.path.exists(f"/sys/class/net/{i}"):
            return i

    return None

def get_ip():

    interfaces = [
        'eth0',
        'enp1s0',
        'end0',
        'wlan0'
    ]

    for ifname in interfaces:

        try:

            s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

            ip = socket.inet_ntoa(
                fcntl.ioctl(
                    s.fileno(),
                    0x8915,
                    struct.pack('256s', ifname[:15].encode())
                )[20:24]
            )

            return ip

        except:
            continue

    return "No Network"

def get_bytes(iface):

    with open(f"/sys/class/net/{iface}/statistics/rx_bytes") as f:
        rx = int(f.read())

    with open(f"/sys/class/net/{iface}/statistics/tx_bytes") as f:
        tx = int(f.read())

    return rx, tx

def format_speed(value):

    if value < 1024:
        return f"{value}K"

    return f"{value/1024:.1f}M"

# =========================================================
# CLEAN EXIT
# =========================================================

def shutdown(sig=None, frame=None):

    oled_clear()
    oled_off()

    bus.close()

    os._exit(0)

signal.signal(signal.SIGTERM, shutdown)
signal.signal(signal.SIGINT, shutdown)

    down = (rx - prev_rx) // 1024
    up = (tx - prev_tx) // 1024

    prev_rx = rx
    prev_tx = tx

    ip = get_ip()

    down_str = format_speed(down)
    up_str = format_speed(up)

    line1 = f"IP: {ip}"
    line2 = f"Net: v{down_str} ^{up_str}"

    clear_buffer()

    draw_text(0, 0, line1)
    draw_text(0, 2, line2)

    flush()
